```

```

To convert using a Node.js script (if no CLI tool is available):
Write a small script `generate-og.mjs`:
```js
// Requires: npm install --save-dev @resvg/resvg-js
import { Resvg } from '@resvg/resvg-js';
import { readFileSync, writeFileSync } from 'fs';
const svg = readFileSync('og-image-template.svg', 'utf8');
const resvg = new Resvg(svg, { fitTo: { mode: 'width', value: 1200 } });
const pngData = resvg.render().asPng();
writeFileSync('og-image.png', pngData);
console.log('og-image.png written');
```
